# Define the Local Network
config firewall address
edit <local-network-object-name>
set subnet <local-network-subnet>
next
end

# Configure the IPsec Tunnel
config vpn ipsec phase1-interface
edit <interface-name>
set interface <interface>
set remote-gw <palo-alto-public-ip>
set ike-version <ike-version>
set proposal <phase1-proposal>
set psksecret <pre-shared-key>
next
end
config vpn ipsec phase2-interface
edit <interface-name>
set phase1name <interface-name>
set proposal <phase2-proposal>
set src-addr-type ip
set src-subnet <local-network-subnet>
set dst-addr-type ip
set dst-subnet <remote-network-subnet>
next
end
config firewall policy
edit <policy-id>
set srcintf <interface-name>
set dstintf
